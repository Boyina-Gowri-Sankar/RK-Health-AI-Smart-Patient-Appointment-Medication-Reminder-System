import streamlit as st
import pandas as pd

st.set_page_config(page_title="RK Health Dashboard", layout="wide")

# Header
st.title("🏥 RK Health Dashboard")
st.write("Patient Reminder & Health Record Management")

# Sidebar
st.sidebar.title("Navigation")
page = st.sidebar.radio(
    "Go to",
    ["Dashboard", "Add Entry", "View Logs", "Health Report"]
)

# Sample data
if "records" not in st.session_state:
    st.session_state.records = []

# Dashboard
if page == "Dashboard":

    col1, col2, col3 = st.columns(3)

    col1.metric("Appointments", len(st.session_state.records))
    col2.metric("Medications", len(st.session_state.records))
    col3.metric("Compliance", "95%")

# Add Entry
elif page == "Add Entry":

    st.subheader("Add Patient Record")

    patient = st.text_input("Patient Name")
    doctor = st.text_input("Doctor Name")
    date = st.date_input("Appointment Date")
    time = st.time_input("Appointment Time")
    medicine = st.text_input("Medication")
    phone = st.text_input("Phone Number")
    notes = st.text_area("Visit Notes")

    if st.button("Save"):

        st.session_state.records.append({
            "Patient": patient,
            "Doctor": doctor,
            "Date": str(date),
            "Time": str(time),
            "Medicine": medicine,
            "Phone": phone,
            "Notes": notes
        })

        st.success("Record Saved Successfully")

# View Logs
elif page == "View Logs":

    st.subheader("Patient Records")

    if st.session_state.records:
        df = pd.DataFrame(st.session_state.records)
        st.dataframe(df, use_container_width=True)
    else:
        st.info("No records available.")

# Health Report
elif page == "Health Report":

    st.subheader("Health Report")

    if st.session_state.records:
        df = pd.DataFrame(st.session_state.records)
        st.table(df)
    else:
        st.info("No report available.")

# AI Summary Section
st.markdown("---")
st.subheader("AI Health Summary")

if st.button("Generate Summary"):
    st.info(
        "Patient attended appointment successfully.\n\n"
        "Recommendation: Continue medication and attend follow-up visit."
    )

# Notification
st.sidebar.success("RK Health System Running")
